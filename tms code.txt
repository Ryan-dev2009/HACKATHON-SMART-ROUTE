  <div id="map"></div>

  <script>
    const map = L.map('map').setView([26.1445, 91.7362], 8);

    L.tileLayer('https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png', {
      maxZoom: 19,
      attribution: '&copy; OpenStreetMap contributors'
    }).addTo(map);

    let control;

    function formatTime(minutes) {
      const hours = Math.floor(minutes / 60);
      const mins = Math.round(minutes % 60);
      return hours > 0 ? `${hours}h ${mins}m` : `${mins} min`;
    }

    document.getElementById('go').addEventListener('click', () => {
      const originText = document.getElementById('origin').value.trim();
      const destText = document.getElementById('destination').value.trim();
      const vehicleType = document.getElementById('vehicleType').value;
      const summary = document.getElementById('summary');
      const details = document.getElementById('details');

      if (!originText || !destText) {
        summary.textContent = 'Please enter both origin and destination.';
        return;
      }

      const [olat, olng] = originText.split(',').map(Number);
      const [dlat, dlng] = destText.split(',').map(Number);

      if (control) map.removeControl(control);

      summary.textContent = 'Calculating route...';
      details.innerHTML = '';

      // Vehicle priority logic
      let color = '#1a73e8';
      let speedFactor = 1.0;
      if (vehicleType === 'emergency') {
        color = 'red';
        speedFactor = 0.6; // 40% faster
      } else if (vehicleType === 'priority') {
        color = 'orange';
        speedFactor = 0.8; // 20% faster
      }

      // Create routing control with built-in instructions (directions panel)
      control = L.Routing.control({
        waypoints: [
          L.latLng(olat, olng),
          L.latLng(dlat, dlng)
        ],
        router: L.Routing.osrmv1({
          serviceUrl: 'https://router.project-osrm.org/route/v1'
        }),
        lineOptions: { styles: [{ color: color, weight: 5 }] },
        collapsible: true,
        showAlternatives: false,
        fitSelectedRoutes: true
      }).addTo(map);

      control.on('routesfound', e => {
        const route = e.routes[0];
        const distanceKm = (route.summary.totalDistance / 1000).toFixed(2);
        let durationMin = route.summary.totalTime / 60;
        durationMin = durationMin * speedFactor;

        summary.innerHTML = `<strong>${distanceKm} km ${formatTime(durationMin)}</strong>`;

        let msg = '';
        if (vehicleType === 'emergency') {
          msg = `<b>Emergency Mode:</b> Traffic lights adjusted, lanes cleared for rapid passage.`;
        } else if (vehicleType === 'priority') {
          msg = `<b>Priority Mode:</b> Dynamic signal control for VIP/Police vehicles.`;
        } else {
          msg = `Normal routing mode active.`;
        }

        details.innerHTML = `<div class="small">${msg}</div>`;
      });
    });
  </script>
